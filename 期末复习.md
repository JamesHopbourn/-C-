### switch 语句
```
#include <stdio.h>

int main() {
  int score;
  char grade;
  do {
    printf("score=");
    scanf("%d", &score);
  } while (score < 0 || score > 100);
  // 此处的 score < 0 || score > 100 写得好
  // 输入错误的情况下不是直接报错，而是允许再次输入
  switch (score / 10) {
  // 此处的 score / 10 写得好
  // 避免了使用分数范围条件，更加简单
  case 10:
  case 9:
    grade = 'A';
    break;
  case 8:
    grade = 'B';
    break;
  case 7:
    grade = 'C';
    break;
  case 6:
    grade = 'D';
    break;
  default:
    grade = 'E';
  }
  printf("%c", grade);
  return 0;
}
```

### 求 2~20 之间的素数
```
#include <math.h>
#include <stdio.h>

int main() {
  int x, i, k;
  for (x = 2; x <= 20; x++) {
    k = sqrt(x);
    // 先设一个 k 等于根号 x
    for (i = 2; i <= k; i++)
      if (x % i == 0)
        break;
    if (i > k)
      printf("%4d", x);
  }
  return 0;
}
```

### 二分查找
```
#include <stdio.h>

int main() {
  int left = 0, right = 9, mid, x;
  int a[10] = {3, 5, 7, 12, 16, 24, 27, 35, 48, 56};
  printf("input x:");
  scanf("%d", &x);
  do {
    mid = (left + right) / 2;
    if (a[mid] == x) {
      printf("%d\n", mid);
      break;
    } else if (a[mid] > x)
      right = mid - 1;
    else
      left = mid + 1;
  } while (left <= right);
  if (left > right) printf("No found");
  return 0;
}
```

### 冒泡排序
```
#include <stdio.h>
#include <time.h>

int main() {
  int i, j, k, t;
  int a[10];
  unsigned m = (unsigned)time(NULL);
  srand(m);
  printf("before:");
  for (i = 0; i < 10; i++)
    a[i] = rand() % 101;
  for (i = 0; i < 10; i++)
    printf("%5d", a[i]);
  for (i = 0; i < 9; i++) {
    for (j = 0; j < 9 - i; j++)
      if (a[j] > a[j + 1]) {
        t = a[j];
        a[j] = a[j + 1];
        a[j + 1] = t;
      }
  }
  printf("\nafter: ");
  for (i = 0; i < 10; i++)
    printf("%5d", a[i]);
  return 0;
}
```
### 选择排序
```
#include <stdio.h>

int main() {
  int i, j, k, t;
  int a[10];
  printf("input a:");
  for (i = 0; i < 10; i++)
    scanf("%d", &a[i]);
  for (i = 0; i < 9; i++) {
    k = i;
    for (j = i + 1; j < 10; j++)
      if (a[j] < a[k])
        k = j;
    if (k != i) {
      t = a[i];
      a[i] = a[k];
      a[k] = t;
    }
  }
  for (i = 0; i < 10; i++)
    printf("%d   ", a[i]);
  return 0;
}
```

```
    return 1;
  else
    return n * f(n - 1);
```

### 最大公约数
```
#include <stdio.h>

int gcd(int a, int b) {
  int r = a % b;
  if (r == 0)
    return b;
  gcd(b, r);
}

int main() {
  int a, b;
  printf("a,b=");
  scanf("%d %d", &a, &b);
  printf("gcd=%d", gcd(a, b));
  return 0;
}
```

### 函数求和
```
#include <stdio.h>

int f(int a[], int n, int *p) {
  int i, s = 0, m;
  for (i = 0; i < n; i++)
    s = s + a[i];
  m = a[0];
  for (i = 0; i < n; i++)
    if (m < a[i])
      m = a[i];
  *p = m;
  return s;
}
int main() {
  int a[10] = {1, 2, 3, 4, 5, 16, 7, 8, 9, 10};
  int sum, max;
  sum = f(a, 10, &max);
  printf("sum=%d max=%d", sum, max);
  return 0;
}
```
